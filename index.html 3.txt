                if(!name || !command){ showNotification('Nombre y comando son requeridos', 'error'); return; }

        function generateInventorySlots(size){
            const container = document.getElementById('minecraftInventory');
            container.innerHTML = '';
            container.style.gridTemplateRows = `repeat(${Math.ceil(size/9)}, 1fr)`;
            
            for(let i = 0; i < size; i++){
                const slot = document.createElement('div');
                slot.className = 'mc-slot';
                slot.dataset.index = i;
                slot.innerHTML = `<span class="slot-number">${i+1}</span>`;
                
                slot.addEventListener('click', () => {
                    selectedSlot = i;
                    document.querySelectorAll('.mc-slot').forEach(s => s.style.borderColor = '');
                    slot.style.borderColor = 'var(--primary)';
                });
                
                slot.addEventListener('dragover', (e) => {
                    e.preventDefault();
                    e.dataTransfer.dropEffect = 'copy';
                });
                
                slot.addEventListener('drop', (e) => {
                    e.preventDefault();
                    if(draggedItem){
                        assignItemToSlot(i, draggedItem);
                        draggedItem = null;
                    }
                });
                
                container.appendChild(slot);
            }
        }
        
        function assignItemToSlot(slotIndex, item){
            const slot = document.querySelector(`.mc-slot[data-index="${slotIndex}"]`);
            if(!slot) return;
            
            guiSlots[slotIndex] = {
                material: item.material,
                name: item.name,
                icon: item.icon,
                productId: null
            };
            
            slot.classList.add('filled');
            slot.innerHTML = `
                <span class="slot-number">${slotIndex+1}</span>
                <div style="font-size:20px">${item.icon}</div>
                <span class="item-name">${item.name}</span>
            `;
        }
        
        function setupGuiEventListeners(){
            document.getElementById('guiType')?.addEventListener('change', async (e) => {
                await loadGuiConfig();
            });
            
            document.getElementById('guiSize')?.addEventListener('change', (e) => {
                const size = parseInt(e.target.value);
                generateInventorySlots(size);
                Object.keys(guiSlots).forEach(key => {

            Object.entries(guiSlots).forEach(([index, item]) => {
                const slot = document.querySelector(`.mc-slot[data-index="${index}"]`);
                if(slot && item){
                    slot.classList.add('filled');
                    slot.innerHTML = `
                        <span class="slot-number">${parseInt(index)+1}</span>
                        <div style="font-size:20px">${item.icon}</div>
                        <span class="item-name">${item.name}</span>
                    `;
                }
            });
        }

        // Load All Orders
        async function loadAllOrders(status = 'all'){
            try{
                if(!currentServerId) return;
                
                const ordersRef = collection(db, 'orders');
                const q = query(ordersRef, where('serverId', '==', currentServerId));
                const snap = await getDocs(q);
                
                let orders = [];
                snap.forEach(d => {
                    const o = d.data();
                    if(status === 'all' || o.status === status || (status === 'pending' && !o.delivered)){
                        orders.push({ id: d.id, ...o });
                    }
                });
                
                orders.sort((a,b) => (b.createdAt?.seconds || 0) - (a.createdAt?.seconds || 0));
                
                const tbody = document.getElementById('allOrdersTable');
                tbody.innerHTML = '';
                
                if(orders.length > 0){
                    orders.forEach(o => {
                        tbody.innerHTML += `
                            <tr>
                                <td><code>#${o.id.substring(0,8)}</code></td>
                                <td>${o.playerName || '-'}</td>
                                <td>${o.productName || '-'}</td>
                                <td><strong>$${(o.price || 0).toFixed(2)}</strong></td>
                                <td>${o.createdAt?.toDate?.().toLocaleDateString('es-ES') || '-'}</td>
                                <td><span class="badge ${o.delivered ? 'success' : o.status === 'cancelled' ? 'danger' : 'pending'}">${o.delivered ? 'Entregado' : o.status === 'cancelled' ? 'Cancelado' : 'Pendiente'}</span></td>
                                <td>
                                    <button class="btn btn-sm btn-secondary" onclick="viewOrder('${o.id}')"><i class="fas fa-eye"></i></button>
                                    ${!o.delivered ? `<button class="btn btn-sm btn-success" onclick="deliverOrder('${o.id}')"><i class="fas fa-check"></i></button>` : ''}
                                    <button class="btn btn-sm btn-danger" onclick="deleteOrder('${o.id}')"><i class="fas fa-trash"></i></button>
                                </td>
                            </tr>
                        `;
                    });
                } else {
                    tbody.innerHTML = '<tr><td colspan="7" class="empty-state"><i class="fas fa-list"></i><p>Sin órdenes</p></td></tr>';
                }
            } catch(e){
                console.error('Error:', e);
            }
        }

        // Chart
        function updateChart(){
            const ctx = document.getElementById('salesChart');
            if(!ctx) return;
            
            const period = parseInt(document.getElementById('chartPeriod').value);
            
            if(salesChart) salesChart.destroy();
            
            const labels = Array.from({length: period}, (_, i) => {
                const d = new Date();
                d.setDate(d.getDate() - (period - i - 1));
                return d.toLocaleDateString('es-ES', {month:'short', day:'numeric'});
            });
            
            const data = Array.from({length: period}, () => Math.floor(Math.random() * 20));
            
            salesChart = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: labels,
                    datasets: [{
                        label: 'Ventas',
                        data: data,
                        borderColor: '#6366f1',
                        backgroundColor: 'rgba(99, 102, 241, 0.1)',
                        fill: true,
                        tension: 0.4,
                        pointBackgroundColor: '#6366f1',
                        pointBorderColor: '#fff',
                        pointBorderWidth: 2
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: { legend: { display: false } },
                    scales: {
                        y: { 
                            beginAtZero: true, 
                            grid: { color: 'rgba(51, 65, 85, 0.3)' },
                            ticks: { color: '#94a3b8' }
                        },
                        x: { 
                            grid: { display: false },
                            ticks: { color: '#94a3b8', maxRotation: 45, minRotation: 45 }
                        }
                    }
                }
            });
        }

        // Export Data
        window.exportData = function(type, format){
            showNotification(`Exportando ${type} en formato ${format.toUpperCase()}...`);
            const data = [{ id: 'demo', name: 'Datos de ejemplo', date: new Date().toISOString() }];
            
            if(format === 'csv'){
                const csv = 'id,name,date\n' + data.map(d => `${d.id},${d.name},${d.date}`).join('\n');
                downloadFile(csv, `${type}_export.csv`, 'text/csv');
            } else {
                const json = JSON.stringify(data, null, 2);
                downloadFile(json, `${type}_export.json`, 'application/json');
            }
            showNotification('Exportación completada');
        };

        function downloadFile(content, filename, mimeType){
            const blob = new Blob([content], { type: mimeType });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = filename;
            a.click();
            URL.revokeObjectURL(url);
        }
